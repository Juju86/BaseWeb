function equalHeight(group) {
  var tallest = 0;
  group.css({ 'height' : 'auto' }).each(function() {
    var thisHeight = $(this).height();
    if(thisHeight > tallest) {
      tallest = thisHeight;
    }
  });
  group.height(tallest);
}
